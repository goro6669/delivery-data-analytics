# assuming this data set came from Jan 2018 temporarily (because it's daylight saving). Going to remove year and month later.
def convertTZ(df, column):
    df[column]=pd.to_datetime('2018-01-'+df[column].astype(str))
    df[column+'_pdt'] = pd.DatetimeIndex(df[column]).tz_localize('UTC').tz_convert('US/Pacific')
    df.reset_index()
    #df[column+'_pdt']=df[column+'_pdt'].dt.time
for i in ['cust_place_time', 'rest_place_time', 'driver_rest_time', 'cust_deliver_time']:
    convertTZ(df, i)
